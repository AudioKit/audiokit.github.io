Auto Wah Wah

One of the most iconic guitar effects is the wah-pedal. This playground runs an audio loop of a guitar through an AKAutoWah node.

{% highlight ruby %}import AudioKitPlaygrounds import AudioKit let file = try AKAudioFile(readFileName: playgroundAudioFiles[0], baseDir: .resources) let player = try AKAudioPlayer(file: file) player.looping = true var wah = AKAutoWah(player) wah.wah = 1 wah.amplitude = 1 AudioKit.output = wah AudioKit.start() player.play() {% endhighlight %}

User Interface Set up

{% highlight ruby %} class PlaygroundView: AKPlaygroundView { override func setup() { addTitle("Auto Wah Wah") addSubview(AKResourcesAudioFileLoaderView( player: player, filenames: playgroundAudioFiles)) addSubview(AKPropertySlider( property: "Wah", value: wah.wah, color: AKColor.green ) { sliderValue in wah.wah = sliderValue }) } } import PlaygroundSupport PlaygroundPage.current.needsIndefiniteExecution = true PlaygroundPage.current.liveView = PlaygroundView() {% endhighlight %}
