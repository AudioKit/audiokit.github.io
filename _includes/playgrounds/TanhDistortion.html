Tanh Distortion

##

{% highlight ruby %}import XCPlayground import AudioKit let bundle = NSBundle.mainBundle() let file = bundle.pathForResource("guitarloop", ofType: "wav") var player = AKAudioPlayer(file!) player.looping = true var distortion = AKTanhDistortion(player) {% endhighlight %}

Set the parameters here

{% highlight ruby %}distortion.pregain = 1.0 distortion.postgain = 1.0 distortion.postiveShapeParameter = 1.0 distortion.negativeShapeParameter = 1.0 AudioKit.output = distortion AudioKit.start() player.play() {% endhighlight %}
