High Pass Filter Operation

{% highlight ruby %}import XCPlayground import AudioKit {% endhighlight %}

Filter setup

{% highlight ruby %}let halfPower = AKOperation.sineWave(frequency: 0.2).scale(minimum: 12000, maximum: 100) let filter = AKOperation.input.highPassFilter(halfPowerPoint: halfPower) {% endhighlight %}

Noise Example

{% highlight ruby %}// Bring down the amplitude so that when it is mixed it is not so loud let whiteNoise = AKWhiteNoise(amplitude: 0.1) let filteredNoise = AKOperationEffect(whiteNoise, operation: filter) {% endhighlight %}

Music Example

{% highlight ruby %}let file = try AKAudioFile(readFileName: "mixloop.wav", baseDir: .Resources) let player = try AKAudioPlayer(file: file) player.looping = true let filteredPlayer = AKOperationEffect(player, operation: filter) {% endhighlight %}

Mixdown and playback

{% highlight ruby %}let mixer = AKDryWetMixer(filteredNoise, filteredPlayer, balance: 0.5) AudioKit.output = mixer AudioKit.start() whiteNoise.start() player.play() XCPlaygroundPage.currentPage.needsIndefiniteExecution = true {% endhighlight %}
